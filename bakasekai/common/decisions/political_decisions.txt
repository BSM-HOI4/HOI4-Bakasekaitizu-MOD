			check_variable = { num_owned_states > 1 }

		cost = 50
		
		complete_effect = {
			if = {
				limit = {
					has_government = democratic_ideology
				}
				set_variable = {
					var = civil_war_size_var
					value = party_popularity@democratic_ideology
				}
			}
			if = {
				limit = {
					has_government = communism_ideology
				}
				set_variable = {
					var = civil_war_size_var
					value = party_popularity@communism_ideology
				}
			}
			if = {
				limit = {
					has_government = neutrality_ideology
				}
				set_variable = {
					var = civil_war_size_var
					value = party_popularity@neutrality_ideology
				}
			}
			subtract_from_variable = {
				var = civil_war_size_var
				value = army_support_var
			}
			if = {
				limit = {
					check_variable = {
						var = civil_war_size_var
						value = 0.3
						compare = less_than
					}
				}
				set_variable = {
					var = civil_war_size_var
					value = 0.3
				}
			}
			start_civil_war = {
				ruling_party = fascism_ideology
				ideology = ROOT
				size = civil_war_size_var
				keep_unit_leaders_trigger = {
					has_trait = hidden_sympathies
				}
			}
			clr_country_flag = preparation_for_fascism_civil_war
			clr_country_flag = military_support_for_fascism_civil_war
			clr_country_flag = civil_support_for_fascism_civil_war

			factor = 0
