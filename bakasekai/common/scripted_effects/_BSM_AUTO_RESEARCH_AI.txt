land_auto_researcher_mother = {
	IF = {
		limit = {
			check_variable = {
				RIW > RISW
			}
		}
		land_auto_researcher_1 = yes
	}
	else_if = {
		limit = {
			check_variable = {
				RIW < RISW
			}
		}
		lar_support_weapons_1 = yes
	}
}
